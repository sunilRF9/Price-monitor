#front end
docker run -d \
     -p 6969:8081 \
     -e ME_CONFIG_MONGODB_ADMINUSERNAME=admin \
     -e ME_CONFIG_MONGODB_ADMINPASSWORD=password \
     --net mongo-network \
     --name mongo-express \
     -e ME_CONFIG_MONGODB_SERVER=mongodb \
     mongo-express

#backend server
docker run -d \
     -p 27017:27017 \
     -e MONGO_INITDB_ROOT_USERNAME=admin \
     -e MONGO_INITDB_ROOT_PASSWORD=password \
     --name mongodb \
     --net mongo-network \
     mongo
